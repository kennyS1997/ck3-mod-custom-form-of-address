# Scope: Title
VNTL_is_vanity_title = {
	is_target_in_global_variable_list = {
		name = VNTL_vanity_titles
		target = THIS
	}
}

# Scope: Character
VNTL_has_vanity_title = {
	any_held_title = {
		VNTL_is_vanity_title = yes
	}
}

# Scope: Title
VNTL_has_current_bound_vt = {
	AND = {
		has_variable = VNTL_bound_vanity_title
		holder = var:VNTL_bound_vanity_title.holder
	}
}

# Scope:        Title - Vanity Title
# TARGET_TITLE: Title - Title to check binding with (should generally be someone's primary)
VNTL_is_vt_bound_to_title = {
	AND = {
		has_variable = VNTL_bound_primary_title
		var:VNTL_bound_primary_title = $TARGET_TITLE$
	}
}

# Scope: Title - Vanity Title
VNTL_is_vt_bound = {
	has_variable = VNTL_bound_primary_title
}

# Scope: Title - Vanity Title
VNTL_is_vt_unbound = {
	NOT = { VNTL_is_vt_bound = yes }
}
